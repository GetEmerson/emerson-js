$.fn.sink



$(target).sink()

             		                            
  $.fn.sink = function() {
    _.each(this, function(e) {
      var elem = $(e);
      var key  = elem.data('sink');

      if(key) {
        _before.apply(elem);
        elem.replaceAll('[data-sink="' + key + '"]', 'body');
        _after.apply(elem);
      }
    });

    return this;
  };

'before' callbacks
             		                            
  var _before = function(callback) {
    var view;
    if(callback) {
      _before.list.push(callback);
    }
    else {
      view = this;
      _.each(_before.list, function(fn) {
        fn(view);
      });
    }
  };
  _.extend(_before,  { list : [] });
